# create-plugin-project Rules

Rules governing standalone plugin projects generated by this skill. Apply all rules silently. Flag conflicts between user requirements and these rules before generating code.

---

## 1. Project Directory Structure

```
{project-name}/
├── src/
│   └── plugins/
│       └── {plugin-name}/
│           ├── commands/
│           │   └── {command-name}/
│           │       ├── handler.ts
│           │       ├── input.ts
│           │       ├── output.ts
│           │       ├── types.ts        # only if Pattern B with complex intermediate types
│           │       └── index.ts
│           ├── __tests__/unit/
│           │   ├── {command-name}.test.ts
│           │   └── helpers/
│           │       ├── mocks.ts
│           │       └── fixtures.ts
│           ├── hooks/                  # only if plugin defines its own hooks
│           ├── services/               # only if 2+ commands share logic that depends on hiero-cli services
│           │   ├── {name}.service.interface.ts
│           │   └── {name}.service.ts
│           ├── utils/                  # pure functions only — no dependency injection
│           ├── manifest.ts
│           ├── index.ts
│           ├── schema.ts               # only if plugin persists data to state
│           └── constants.ts            # only if plugin-scoped constants used in 2+ files
├── package.json
├── tsconfig.base.json
├── tsconfig.json
├── tsconfig.test.json
├── jest.config.js
├── jest.unit.config.js
├── eslint.config.js
├── .gitignore
└── README.md
```

## 3. Import Convention

**Critical rule:** All hiero-cli types, base classes, schemas, and error classes import from `@hiero-ledger/hiero-cli`. Never use `@/core` or any internal path from the hiero-cli repository.

```typescript
// types
import type {
  Command,
  CommandHandlerArgs,
  CommandResult,
  PluginManifest,
  CoreApi,
} from '@hiero-ledger/hiero-cli';

// base result interfaces for Pattern B types.ts
import type {
  BaseNormalizedParams,
  BaseBuildTransactionResult,
  BaseSignTransactionResult,
  BaseExecuteTransactionResult,
} from '@hiero-ledger/hiero-cli';

// classes
import { BaseTransactionCommand } from '@hiero-ledger/hiero-cli';

// errors
import {
  ValidationError,
  ConfigurationError,
  NotFoundError,
  InternalError,
  TransactionError,
} from '@hiero-ledger/hiero-cli';

// enums
import { OptionType } from '@hiero-ledger/hiero-cli';

// Zod schemas re-exported from the package
import {
  EntityIdSchema,
  NetworkSchema,
  TransactionIdSchema,
  MemoSchema,
  AliasNameSchema,
  KeySchema,
  AmountInputSchema,
  EvmAddressSchema,
} from '@hiero-ledger/hiero-cli';
```

Project-internal imports use the `@/` alias (mapped to `src/`):

```typescript
import { MyPluginCreateInputSchema } from '@/plugins/my-plugin/commands/create/input';
```

## 4. Handler Patterns

### Pattern A: Simple Command (no transaction lifecycle)

Use for: read-only, list, view, config commands.

```typescript
import type {
  Command,
  CommandHandlerArgs,
  CommandResult,
} from '@hiero-ledger/hiero-cli';
import { MyPluginListInputSchema } from './input';

export class MyPluginListCommand implements Command {
  async execute(args: CommandHandlerArgs): Promise<CommandResult> {
    const validArgs = MyPluginListInputSchema.parse(args.args);
    // IMPLEMENT
    return { result: {} };
  }
}

export async function myPluginList(
  args: CommandHandlerArgs,
): Promise<CommandResult> {
  return new MyPluginListCommand().execute(args);
}
```

### Pattern B: Transaction Command (build → sign → execute lifecycle)

Use for: commands that submit Hedera transactions.

Generate `types.ts` for every Pattern B command using the base interfaces:

```typescript
// types.ts
import type {
  BaseBuildTransactionResult,
  BaseExecuteTransactionResult,
  BaseNormalizedParams,
  BaseSignTransactionResult,
} from '@hiero-ledger/hiero-cli';

export interface CreateNormalisedParams extends BaseNormalizedParams {
  // add command-specific normalized fields here
}

export interface CreateBuildResult extends BaseBuildTransactionResult {}
export interface CreateSignResult extends BaseSignTransactionResult {}
export interface CreateExecuteResult extends BaseExecuteTransactionResult {}
```

```typescript
// handler.ts

  CommandHandlerArgs,
  CommandResult,
} from '@hiero-ledger/hiero-cli';
import { BaseTransactionCommand } from '@hiero-ledger/hiero-cli';
import type {
  CreateNormalisedParams,
  CreateBuildResult,
  CreateSignResult,
  CreateExecuteResult,
} from './types';
import { MyPluginCreateInputSchema } from './input';

export class MyPluginCreateCommand extends BaseTransactionCommand<
  CreateNormalisedParams,
  CreateBuildResult,
  CreateSignResult,
  CreateExecuteResult
> {
  async normalizeParams(
    args: CommandHandlerArgs,
  ): Promise<CreateNormalisedParams> {
    const validArgs = MyPluginCreateInputSchema.parse(args.args);
    // IMPLEMENT
    return {} as CreateNormalisedParams;
  }

  async buildTransaction(
    args: CommandHandlerArgs,
    params: CreateNormalisedParams,
  ): Promise<CreateBuildResult> {
    // IMPLEMENT
    return {} as CreateBuildResult;
  }

  async signTransaction(
    args: CommandHandlerArgs,
    params: CreateNormalisedParams,
    buildResult: CreateBuildResult,
  ): Promise<CreateSignResult> {
    // IMPLEMENT
    return {} as CreateSignResult;
  }

  async executeTransaction(
    args: CommandHandlerArgs,
    params: CreateNormalisedParams,
    buildResult: CreateBuildResult,
    signResult: CreateSignResult,
  ): Promise<CreateExecuteResult> {
    // IMPLEMENT
    return {} as CreateExecuteResult;
  }

  async outputPreparation(
    args: CommandHandlerArgs,
    params: CreateNormalisedParams,
    buildResult: CreateBuildResult,
    signResult: CreateSignResult,
    executeResult: CreateExecuteResult,
  ): Promise<CommandResult> {
    // IMPLEMENT
    return { result: {} };
  }
}

export async function myPluginCreate(
  args: CommandHandlerArgs,
): Promise<CommandResult> {
  return new MyPluginCreateCommand().execute(args);
}
```

Rules:

- Always export both the class AND the function wrapper
- Use Pattern A by default; Pattern B only when the command submits a transaction
- `normalizeParams` is where input schema parsing happens — not in `buildTransaction`
- Never put business logic in the function wrapper
- Use real imports everywhere — never write placeholder comments like `// TODO: import X`
- Mark every unimplemented handler body with `// IMPLEMENT` so the developer knows exactly where to add logic

## 5. Input Schemas

```typescript
import { z } from 'zod';
import { AliasNameSchema, EntityIdSchema } from '@hiero-ledger/hiero-cli';

export const MyPluginCreateInputSchema = z.object({
  name: AliasNameSchema.describe('Name for the resource'),
  id: EntityIdSchema.describe('Hedera entity ID'),
});

export type MyPluginCreateInput = z.infer<typeof MyPluginCreateInputSchema>;
```

Rules:

- Reuse schemas from `@hiero-ledger/hiero-cli`: `EntityIdSchema`, `MemoSchema`, `KeySchema`, `AmountInputSchema`, `AliasNameSchema`, `NetworkSchema`, `TransactionIdSchema`, `EvmAddressSchema`, `EmptyOrNullableKeyListSchema`
- Export both the schema and the inferred TypeScript type
- Use `.describe()` on every field — it powers CLI help text
- Use `z.superRefine()` for cross-field validation (mutually exclusive options, conditional requirements)

## 6. Output Schemas

```typescript
import { z } from 'zod';
import {
  EntityIdSchema,
  NetworkSchema,
  TransactionIdSchema,
} from '@hiero-ledger/hiero-cli';

export const MyPluginCreateOutputSchema = z.object({
  id: EntityIdSchema,
  network: NetworkSchema,
  transactionId: TransactionIdSchema,
});

export type MyPluginCreateOutput = z.infer<typeof MyPluginCreateOutputSchema>;

export const MY_PLUGIN_CREATE_TEMPLATE = `
✅ Resource created: {{id}}
   Network: {{network}}
   Transaction: {{hashscanLink transactionId "transaction" network}}
`.trim();
```

Rules:

- Export both the schema and the inferred type
- Export a Handlebars template constant
- Use `{{hashscanLink entityId "account"|"transaction"|"token" network}}` for explorer links
- Use `{{#if field}}...{{/if}}` for optional fields; `{{#each items}}...{{/each}}` for arrays
- `✅` prefix for create/update/delete commands; no prefix for list/view

## 7. Manifest Structure

```typescript
import type { PluginManifest } from '@hiero-ledger/hiero-cli';
import { OptionType } from '@hiero-ledger/hiero-cli';
import { MyPluginCreateCommand } from './commands/create/handler';
import {
  MyPluginCreateOutputSchema,
  MY_PLUGIN_CREATE_TEMPLATE,
} from './commands/create/output';

export const MyPluginPluginManifest: PluginManifest = {
  name: 'my-plugin',
  version: '1.0.0',
  displayName: 'My Plugin',
  description: 'What this plugin does',
  commands: [
    {
      name: 'create',
      summary: 'Create a resource',
      description: 'Longer description of the create command',
      options: [
        {
          name: 'name',
          short: 'n',
          type: OptionType.STRING,
          required: true,
          description: 'Name of the resource',
        },
      ],
      command: new MyPluginCreateCommand(),
      output: {
        schema: MyPluginCreateOutputSchema,
        humanTemplate: MY_PLUGIN_CREATE_TEMPLATE,
      },
    },
  ],
};

export default MyPluginPluginManifest;
```

Rules:

- Every option must have: `name`, `short` (single char), `type`, `required`, `description`
- `short` must not conflict with reserved chars: `h`, `V`, `N`, `P`; batchify hook reserves `B`; scheduled hook reserves `X`
- Prefer `command: new MyPlugin{Name}Command()` (class-based); use `handler: myPluginFn` only for the simplest commands
- Include `registeredHooks` only for transaction commands opting into cross-plugin hooks (batchify, scheduled)
- Include `requireConfirmation: true` for destructive commands (delete, clear, wipe)
- Always include `export default MyPluginPluginManifest` as the final line — `loadPluginManifest` looks for the default export first

## 8. Error Handling

Import all error classes from `@hiero-ledger/hiero-cli`. Never throw raw `Error`.

| Situation                               | Error Class                |
| --------------------------------------- | -------------------------- |
| Invalid user input                      | `ValidationError`          |
| Missing config (operator not set, etc.) | `ConfigurationError`       |
| Resource not found                      | `NotFoundError`            |
| Unexpected internal failure             | `InternalError`            |
| Insufficient balance                    | `InsufficientBalanceError` |
| Transaction failure                     | `TransactionError`         |

## 9. Hook Registration

```typescript
registeredHooks: [
  { hook: 'batchify-set-batch-key', phase: 'preSignTransaction' },
  { hook: 'scheduled', phase: 'preSignTransaction' },
  { hook: 'batchify-add-transaction', phase: 'preExecuteTransaction' },
],
```

Rules:

- Only transaction commands (Pattern B) register for `batchify-*` and `scheduled` hooks
- Never register hooks on read-only commands
- Hooks are declared on the command spec in the manifest — not inside the handler

## 11. Unit Tests

### mocks.ts

```typescript
import type { CoreApi, CommandHandlerArgs } from '@hiero-ledger/hiero-cli';
import type { Logger } from '@hiero-ledger/hiero-cli';

export const makeLogger = (): jest.Mocked<Logger> => ({
  info: jest.fn(),
  error: jest.fn(),
  debug: jest.fn(),
  warn: jest.fn(),
  setLevel: jest.fn(),
});

// Include logger, state, config as part of CoreApi — they are no longer top-level CommandHandlerArgs fields
export const makeApiMock = (
  overrides?: Partial<jest.Mocked<CoreApi>>,
): jest.Mocked<CoreApi> =>
  ({
    logger: makeLogger(),
    ...overrides,
  }) as unknown as jest.Mocked<CoreApi>;

// Note: state, logger, and config are accessed via args.api.* — not top-level fields
export const makeCommandArgs = (params: {
  api: jest.Mocked<CoreApi>;
  args?: Record<string, unknown>;
}): CommandHandlerArgs => ({
  args: { ...(params.args ?? {}) },
  api: params.api,
  hooks: new Map(),
});

// Add one make{Plugin}{Command}Args factory per command — pre-filled with valid defaults
export const makeMyPluginCreateArgs = (params: {
  api: jest.Mocked<CoreApi>;
  args?: Partial<{ name: string }>;
}): CommandHandlerArgs =>
  makeCommandArgs({
    api: params.api,
    args: {
      name: 'my-resource',
      ...params.args,
    },
  });
```

Rules:

- All mock factories use `jest.fn()` — never real implementations
- Use `as unknown as jest.Mocked<T>` when casting partial mocks
- Support `overrides` on API-level mocks
- Never import from `fixtures.ts` in `mocks.ts`
- `state`, `logger`, and `config` are no longer top-level `CommandHandlerArgs` fields — they are properties of `CoreApi` and accessed via `args.api.*` in handlers

### fixtures.ts

Contains typed constant data. Minimum per command:

- One valid input object (happy path)
- One invalid input object (missing required field)
- One invalid input object (wrong format)
- Factory functions (`make*`) that accept `Partial<T>` overrides

### {command}.test.ts

Minimum coverage per command: one happy-path test and one validation-error test.

```typescript
describe('{PluginName} {command} command', () => {
  let api: jest.Mocked<CoreApi>;

  beforeEach(() => {
    api = makeApiMock();
  });

  it('returns expected output on valid input', async () => {
    const args = makeMyPluginCreateArgs({ api });
    const result = await myPluginCreate(args);
    expect(result.result).toMatchObject({
      /* expected shape */
    });
  });

  it('throws ValidationError when name is missing', async () => {
    const args = makeMyPluginCreateArgs({
      api,
      args: { name: undefined },
    });
    await expect(myPluginCreate(args)).rejects.toThrow(ValidationError);
  });
});
```

Rules:

- Use `beforeEach` to reset mocks — never share mock state between tests; recreate `api = makeApiMock()` there
- Arrange / Act / Assert order in every `it` block
- Import all test data from `fixtures.ts` — never hardcode strings inline in assertions
- One `describe` block per command file, named `'{PluginName} {command} command'`
- `logger` is no longer a separate variable — it is part of `api` (`api.logger`)

## 12. Project Config Files

### Strategy: copy from hiero-ledger/hiero-cli

For all config files **except `package.json`**, copy the current version directly from the `hiero-ledger/hiero-cli` repository rather than generating from a template. This guarantees compatibility with the version of `@hiero-ledger/hiero-cli` being used and avoids the config drifting out of sync as the project evolves.

Files to copy verbatim:

| File                | Source path in hiero-ledger/hiero-cli |
| ------------------- | ------------------------------------- |
| `tsconfig.base.json`  | `tsconfig.base.json`                |
| `tsconfig.json`       | `tsconfig.json`                     |
| `tsconfig.test.json`  | `tsconfig.test.json`                |
| `jest.config.js`      | `jest.config.js`                    |
| `jest.unit.config.js` | `jest.unit.config.js`               |
| `eslint.config.js`    | `eslint.config.js`                  |

If the user supplied a local `.tgz`, extract the source config files from it. Otherwise fetch the files from the published repository at the same tag as the resolved npm version.

## 13. Plugin Services Pattern

Plugin-internal logic that is **reused across 2+ commands and depends on `@hiero-ledger/hiero-cli` services** belongs in a plugin service — a class with an interface under `src/plugins/{name}/services/`.

**Decision rule:**

| Logic                                         | Location                  |
| --------------------------------------------- | ------------------------- |
| Reused across commands + needs hiero-cli      | `services/` (this section)|
| Reused across commands + pure (no deps)       | `utils/`                  |
| Single command only + needs hiero-cli         | inline via `args.api.X`   |

### File layout

```
src/plugins/{name}/services/
├── {thing}.service.interface.ts   # interface only
└── {thing}.service.ts             # implementation
```

### Interface

```typescript
// my-plugin-state.service.interface.ts
import type { MyPluginData } from '@/plugins/my-plugin/schema';

export interface MyPluginStateService {
  get(name: string, network: string): MyPluginData;
  save(key: string, data: MyPluginData): void;
}
```

### Implementation

```typescript
// my-plugin-state.service.ts
import type { CoreApi } from '@hiero-ledger/hiero-cli';

export class MyPluginStateServiceImpl implements MyPluginStateService {
  constructor(
    private readonly state: CoreApi['state'],
    private readonly logger: CoreApi['logger'],
  ) {}

  get(name: string, network: string): MyPluginData {
    // IMPLEMENT
  }

  save(key: string, data: MyPluginData): void {
    // IMPLEMENT
  }
}
```

Rules:

- Constructor accepts **only concrete services extracted from `CoreApi`** — never the full `CoreApi` object
- All constructor parameters are declared `private readonly`
- Interface lives in a separate `*.service.interface.ts` file

### Wiring in the wrapper function

The exported `async function` in `handler.ts` is the **only place** where `args.api` is consumed to construct plugin services:

```typescript
export class MyPluginCreateCommand extends BaseTransactionCommand<...> {
  constructor(
    private readonly myState: MyPluginStateService,
  ) { super('my-plugin_create'); }

  async normalizeParams(args: CommandHandlerArgs): Promise<...> {
    const data = this.myState.get(...);
    // IMPLEMENT
  }
  // buildTransaction / signTransaction / executeTransaction use args.api directly
}

export async function myPluginCreate(args: CommandHandlerArgs): Promise<CommandResult> {
  const { api } = args;
  const myState = new MyPluginStateServiceImpl(api.state, api.logger);
  return new MyPluginCreateCommand(myState).execute(args);
}
```

**Mental model:** `this.X` → plugin service; `args.api.X` → hiero-cli Core service.

### Testing plugin services

Test each service in isolation. Mock only its constructor dependencies — not the full API:

```typescript
describe('MyPluginStateService', () => {
  it('returns stored data', () => {
    const stateMock = {
      get: jest.fn().mockReturnValue(mockMyPluginData),
    } as unknown as jest.Mocked<CoreApi['state']>;
    const loggerMock = makeLogger();
    const service = new MyPluginStateServiceImpl(stateMock, loggerMock);

    const result = service.get('my-resource', 'testnet');

    expect(result).toEqual(mockMyPluginData);
    expect(stateMock.get).toHaveBeenCalledWith(expect.any(String));
  });
});
```
